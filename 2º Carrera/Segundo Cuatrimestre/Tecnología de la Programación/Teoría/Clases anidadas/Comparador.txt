package comparator;

import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;

//Ejemplo para ver como se crea un comparador y como lo utilizamos 
//para ordenar los elementos de un arrayList con el método sort(_)

public class EjemploComparator {
	static private List<String> miArrayList;
	static private Comparator<String> cmpString = new Comparator<String>() {

		public int compare(String s1, String s2) {
			if (s1.length() < s2.length()) {
				return -1;
			} else if (s1.length() > s2.length()) {
				return 1;
			} else {
				return 0;
			}
		}
	};

	public static void main(String[] args) {
		miArrayList = new ArrayList<String>();

		miArrayList.add("Jacinto");
		miArrayList.add("Luis");
		miArrayList.add("Federico");
		miArrayList.add("Miguel Angel");

		// sin ordenar, tal y como los introduje
		System.out.println("Sin ordenar: " + miArrayList);

		// ordenando tirando del compareTo de la clase
		Collections.sort(miArrayList);
		System.out.println("Ordenando por orden alfabético: " + miArrayList);

		// ordeno con mi comparador
		miArrayList.sort(cmpString);
		System.out.println("Ordenando con mi comparador: " + miArrayList);

	}
}
